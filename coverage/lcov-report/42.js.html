const M = 1000000007n;
const MAX = 200005;
 
function powmod(a, b) {
    let res = 1n;
    a %= M;
    while (b > 0n) {
        if (b & 1n) res = (res * a) % M;
        a = (a * a) % M;
        b >>= 1n;
    }
    return res;
}
